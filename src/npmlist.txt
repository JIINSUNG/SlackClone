@@@@@@@@@@ install @@@@@@@@@@@@
1. npx create-react-app slack-clone --template redux
2. npm install -g yarn
3. yarn add react-router-dom
4. yarn add @material-ui/core
5. yarn add styled-components (package.json, add resolution)
  "resolutions": {
    "styled-components" : "^5"
  }
6. yarn add @mui/icons-material
7. yarn add firebase
8. yarn add react-firebase-hooks
9. yarn add react-spinkit
10. npm install -g firebase-tools
